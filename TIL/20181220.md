## react에서 map 함수에서 br태그 렌더링 하기

```js
  const a = ["안녕 나는 구보현이야. \n 안녕 안녕 \n 안녕하세요."];
    return (
      <div>
        {a.map(text =>
          text.split("\n").map(line => {
            return (
              <span >
                {line}
                <br />
              </span>
            );
          })
        )}
      </div>
    );
  }
```
